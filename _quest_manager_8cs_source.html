102 public void NewQuest(Quests _Quest)

130 public void AcceptQuest()
131 {
132 //Turns off the accept quest UI
133 QuestAcceptWindow.SetActive(false);
134 //Create a new button for the journal UI
135 GameObject NewButton = (GameObject)Instantiate(QuestButtonPrefab, QuestButtonParent.transform);
136
137 //Add to list of buttons and give it a listener to a method that is called when clicked
138 Buttons.Add(NewButton);
139 int index = QuestList.Count - 1;
140 NewButton.GetComponentInChildren<TextMeshProUGUI>().text = QuestList[index].GetTitle();
141 NewButton.GetComponent<Button>().onClick.AddListener(() => UpdateUI(index));
142
143 //Set the buttons active states to be true as default
144 foreach (Quests Quest in QuestList)

146 if (Quest.QuestIsActive == false)
147 {
148 Quest.QuestIsActive = true;

163 public void CompleteTask(GoalType Type)
164 {
165 for(int i = 0; i < QuestList.Count; i++)
166 {
167 if (QuestList[i].Tag == Type && QuestList[i].QuestIsActive == true)
168 {
169 if (!QuestList[i].Evaluate())
170 {
171 QuestList[i].CurrentAmount++;
172 if (QuestList[i].Evaluate())
173 {
174 GameManager.Instance.UpdateMoney(QuestList[i].Reward);
175 QuestList[i].QuestIsActive = false;
176 GameObject temp = Buttons[i];
177 Buttons.Remove(Buttons[i]);
178 Destroy(temp);
179 }
180 }
181 }
182 }
