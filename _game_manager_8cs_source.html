26 public static GameManager Instance
27 {
28 get
29 {
30 if (m_ShuttingDown)
31 {
32 Debug.LogWarning("[Singleton] Instance '" + typeof(GameManager) +
33 "' already destroyed. Returning null.");
34 return null;
35 }
36 lock (m_Lock)
37 {
38 if (m_Instance == null)
39 {
40 // Search for existing instance.
41 m_Instance = (GameManager)FindObjectOfType(typeof(GameManager));
42 // Create new instance if one doesn't already exist.
43 if (m_Instance == null)
44 {
45 // Need to create a new GameObject to attach the singleton to.
46 var singletonObject = new GameObject();
47 m_Instance = singletonObject.AddComponent<GameManager>();
48 singletonObject.name = typeof(GameManager).ToString() + " (Singleton)";
49 // Make instance persistent.
50 DontDestroyOnLoad(singletonObject);
51 }
52 }
53 return m_Instance;
54 }
55 }
