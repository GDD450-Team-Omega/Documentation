146 /*
147 * moves collider based on given positions
148 *
149 * Parameters: RightPos | Right position of line, LeftPos | Left position of line , col | collider to move
150 * Return: void
151 */
152 private void MoveCollider(Vector3 RightPos, Vector3 LeftPos, BoxCollider col)
153 {
154 col.size = new Vector3(Vector3.Distance(LeftPos, RightPos), 10f, 5f); // size of collider is set where X is length of line, Y is width of line, Z will be set as per requirement
155 col.transform.localPosition = (RightPos + LeftPos) / 2;
156 // Following lines calculate the angle between startPos and endPos
157 float angle = (Mathf.Abs(RightPos.y - LeftPos.y) / Mathf.Abs(RightPos.x - LeftPos.x));
158 if ((RightPos.y < LeftPos.y && RightPos.x > LeftPos.x) || (LeftPos.y < RightPos.y && LeftPos.x > RightPos.x))
159 {
160 angle *= -1;
161 }
162 col.transform.eulerAngles = new Vector3(0, 0, Mathf.Rad2Deg * Mathf.Atan(angle));
163 }

284 private void Update()
285 {
286 // empty line renderer to check against
287 LineRenderer EmptyLineRenderer = new LineRenderer();
288
289 // Disables the current line when gets to the other side

299
300 // detects when mouse is clicked
301 if (Input.GetMouseButtonDown(0))
302 {
303 MousePressed = true;
304 }
305
306 // detects when mouse is released
307 if (Input.GetMouseButtonUp(0))
308 {
309 MousePressed = false;
310 }
311
312 if (CurrentLine != EmptyLineRenderer && MousePressed == false)
313 {
314 if (Vector3.Distance(CurrentLine.GetPosition((int)CurrentSide), PreviousLinePos) > 25)
315 {
316 CurrentLine.SetPosition((int)CurrentSide, PreviousLinePos);
317 ResetMouseInfo();
318 }
319 }
320
321 // changes the position of the current line to the mouse position
322 if (MousePressed && CurrentLine != EmptyLineRenderer)
323 {
324 MoveLine();
